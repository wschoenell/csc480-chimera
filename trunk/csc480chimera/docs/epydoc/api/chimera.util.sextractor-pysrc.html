 73  """ 
 74  A wrapper for SExtractor 
 75   
 76  A wrapper for SExtractor, the Source Extractor. 
 77  by Laurent Le Guillou 
 78  version: 1.15 - last modified: 2005-07-06 
 79   
 80  This wrapper allows you to configure SExtractor, run it and get 
 81  back its outputs without the need of editing SExtractor 
 82  configuration files. by default, configuration files are created 
 83  on-the-fly, and SExtractor is run silently via python. 
 84   
 85  Tested on SExtractor versions 2.2.1 and 2.3.2. 
 86   
 87   
 88  Example of use: 
 89   
 90  ----------------------------------------------------------------- 
 91   
 92      import sextractor 
 93   
 94      # Create a SExtractor instance 
 95      sex = sextractor.SExtractor() 
 96   
 97      # Modify the SExtractor configuration 
 98      sex.config['GAIN'] = 0.938 
 99      sex.config['PIXEL_SCALE'] = .19 
100      sex.config['VERBOSE_TYPE'] = "FULL" 
101      sex.config['CHECKIMAGE_TYPE'] = "BACKGROUND" 
102   
103      # Add a parameter to the parameter list 
104      sex.config['PARAMETERS_LIST'].append('FLUX_BEST') 
105   
106      # Lauch SExtractor on a FITS file 
107      sex.run("nf260002.fits") 
108   
109      # Read the resulting catalog [first method, whole catalog at once] 
110      catalog = sex.catalog() 
111      for star in catalog: 
112          print star['FLUX_BEST'], star['FLAGS'] 
113          if (star['FLAGS'] & sextractor.BLENDED): 
114              print "This star is BLENDED" 
115   
116      # Read the resulting catalog [second method, whole catalog at once] 
117      catalog_name = sex.config['CATALOG_NAME'] 
118      catalog_f = sextractor.open(catalog_name) 
119      catalog = catalog_f.readlines() 
120      for star in catalog: 
121          print star['FLUX_BEST'], star['FLAGS'] 
122          if (star['FLAGS'] & sextractor.BLENDED): 
123              print "This star is BLENDED" 
124      catalog_f.close() 
125   
126      # Read the resulting catalog [third method, star by star] 
127      catalog_name = sex.config['CATALOG_NAME'] 
128      catalog_f = sextractor.open(catalog_name) 
129      star = catalog_f.readline() 
130      while star: 
131          print star['FLUX_BEST'], star['FLAGS'] 
132          if (star['FLAGS'] & sextractor.BLENDED): 
133              print "This star is BLENDED" 
134          star = catalog_f.readline() 
135      catalog_f.close() 
136   
137      # Removing the configuration files, the catalog and 
138      # the check image 
139      sex.clean(config=True, catalog=True, check=True)
